Class Phalcon\CLI\Dispatcher

Namespace: Phalcon\CLI
Implements: Phalcon\Events\EventsAwareInterface, Phalcon\DI\InjectionAwareInterface, Phalcon\DispatcherInterface,
Extends: Phalcon\Dispatcher
Phalcon\CLI\Dispatcher Dispatching is the process of taking the command-line arguments, extracting the module name, task name, action name, and optional parameters contained in it, and then instantiating a task and calling an action on it.

	$di = new Phalcon\DI();

	$dispatcher = new Phalcon\CLI\Dispatcher();

  $dispatcher->setDI($di);

	$dispatcher->setTaskName('posts');
	$dispatcher->setActionName('index');
	$dispatcher->setParams(array());

	$handle = $dispatcher->dispatch();
